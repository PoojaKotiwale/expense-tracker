// for colosing the popover on defferent events (onclick, scroll, mouseout, cancel)
$('body').on('click', function (e) {
	// hide any open popovers when the anywhere else in the body is clicked
	if (!$('[data-toggle=popover]').is(e.target) && $('[data-toggle=popover]').has(e.target).length === 0 && $('.popover').has(e.target).length === 0) {
		$('[data-toggle=popover]').popover('hide');
	}
});

$(document).ready(function() {
    // Master Checkbox functionality
    $("#masterCheckbox").on('change', function() {
        // Get the checked state of the master checkbox
        var isChecked = $(this).is(':checked');
        
        // Update all USER_DEFINED checkboxes (visible ones only)
        $("#remark_types_dropdown input[type='checkbox']:visible")
            .not("#masterCheckbox")
            .prop('checked', isChecked);
        
        // Update the remark filter based on the current selections
        updateRemarkFilter();
    });

    // Individual checkbox functionality
    $("#remark_types_dropdown input[type='checkbox']").not("#masterCheckbox").on('change', function() {
        // Update the master checkbox state based on individual checkboxes
        updateMasterCheckbox();
        
        // Update the remark filter based on the current selections
        updateRemarkFilter();
    });

    // Function to update the remark filter based on checked boxes
    function updateRemarkFilter() {
        var selectedRemarks = [];
        
        // Collect all checked remark types
        $('#remark_types_dropdown input[type="checkbox"]:checked').not("#masterCheckbox").each(function() {
            var remarkType = $.trim($(this).val());
            if (remarkType) selectedRemarks.push(remarkType);
        });

        // Update the hidden filter field
        $("#filter_remark_type").val(JSON.stringify(selectedRemarks));
        $("#filter_remark_type").attr('remark_type_count', selectedRemarks.length);

        // UI Updates
        if (selectedRemarks.length > 0) {
            $("#remark_filter_div").removeClass("hide");
            $("#history_timeline").html(""); // Clear existing timeline
            $("#more_history_btn").show();
            $("#list_heading").text('Remark Types within ' + selectedRemarks.join(", "));
            
            // Make a SINGLE call to load remarks based on selected filter
            if (from_page == 'ACCOUNT_DETAIL') {
                LoadMoreDetailSysUdates(15, 'history_timeline', 'more_history_btn');
            } else {
                LoadMoreAccountSysUdates(15, 'history_timeline', 'more_history_btn');
            }
        } else {
            $("#remark_filter_div").addClass("hide");
            $("#history_timeline").html('<div class="color-red-dark mt10">No remark type is selected.</div>').addClass("mb15");
            $("#more_history_btn").hide();
        }
    }

    // Update master checkbox based on individual checkbox states
    function updateMasterCheckbox() {
        var $checkboxes = $("#remark_types_dropdown input[type='checkbox']:visible").not("#masterCheckbox");
        var allChecked = $checkboxes.length > 0 && $checkboxes.length === $checkboxes.filter(":checked").length;
        $("#masterCheckbox").prop("checked", allChecked);
    }

    // When showing system updates
    $("#show_sys_update_btn").click(function() {
        $("#sys_updates").val('1');
        $("#show_sys_update_btn").addClass('hide');
        $("#hide_sys_update_btn").removeClass('hide');
        
        // Check all checkboxes (including system ones)
        $("#remark_types_dropdown input[type='checkbox']").prop('checked', true);
        
        updateRemarkFilter();
    });

    // When hiding system updates
    $("#hide_sys_update_btn").click(function() {
        $("#sys_updates").val('0');
        $("#show_sys_update_btn").removeClass('hide');
        $("#hide_sys_update_btn").addClass('hide');
        
        // Uncheck all system remark checkboxes
        $(".sys_remarks input[type='checkbox']").prop('checked', false);
        
        updateRemarkFilter();
        updateMasterCheckbox();
    });
});



$("#remark_types_dropdown li a input").click(function(){
	var remarktype = $.trim($(this).val());
	var index = remark_filter.indexOf(remarktype);
    var remark_filter_count = parseInt($("#filter_remark_type").attr('remark_type_count'));  
	
	if(remarktype != 'MORE'){
	
    	$('#history_timeline').html('');
    	$('#history_timeline').attr('limit_from',0);
    	$('#more_history_btn').show();
    	
        //INCREMENTING THE SELECTED REMARK TYPE COUNTER & INCLUDING IN THE REMARK FILTER ARRAY, IF SHOW ALL SYSTEM UPDATE IS SELECTED
        if($("#hide_sys_update_btn").is(':visible')) {
            $.each($("#remark_types_dropdown li"), function(){
            	var rem_type = $.trim($(this).text());
            	if(rem_type != 'MORE' && rem_type != '' && remarktype != rem_type){ 
                	if($(this).hasClass('sys_remarks')){
                    	if($('#remark_types_dropdown [value="'+rem_type+'"]').is(':checked')) {
                    		var sys_rem_index = remark_filter.indexOf(rem_type);
                    		if(sys_rem_index == -1){
                        		remark_filter.push(rem_type);
            					//remark_filter_count++;
            				}
                    	}
                    }
                }
            	
        	});
        }
    
    
		//IF REMARK TYPE IS CHECKED, INCREMENTING THE SELECTED REMARK TYPE COUNTER & INCLUDING IN THE REMARK FILTER ARRAY 
		if($(this).is(':checked')){
			if(index == -1){
    			remark_filter.push(remarktype);
    			remark_filter_count++;
      			$("#filter_remark_type").attr('remark_type_count',remark_filter_count);
      		}
		}
		//IF REMARK TYPE IS UN CHECKED, DECREMENTING THE SELECTED REMARK TYPE COUNTER & REMOVING IN THE REMARK FILTER ARRAY
		else {
 			if (index !== -1) {
      			remark_filter.splice(index, 1);
  				if(remark_filter_count > 0){
          			remark_filter_count--;
      				$("#filter_remark_type").attr('remark_type_count',remark_filter_count);
      			}
 			}
		}
		
		$("#filter_remark_type").val(JSON.stringify(remark_filter));
		
		//IF SELECTED REMARK TYPE IS NOT EMPTY , LOADING THE REMARKS
    	if(remark_filter_count > 0) {
     		$("#remark_filter_div").removeClass('hide');
     		
     		if(from_page == 'ACCOUNT_DETAIL') {
     			LoadMoreDetailSysUdates(15, 'history_timeline', 'more_history_btn');
     		}
     		else{
     			LoadMoreAccountSysUdates(15, 'history_timeline', 'more_history_btn');
     		}
     	}
     	else {
     		$("#history_timeline").html('<div class="color-red-dark mt10">No remark type is selected.</div>').addClass('mb15');
     		$("#more_history_btn").hide();
     		$("#remark_filter_div").addClass('hide');
     	}
    
    	$("#list_heading").html('Remark Types within '+ remark_filter.join(", "));

	}
});

function ShowSystemUpdates() {
	HidePopover();
	$("#sys_updates").val('1');
	$("#filter_remark_type").val('');
	$('#history_timeline').html('');
	$('#history_timeline').attr('limit_from',0);
	$('#more_history_btn').show();

	$("#show_sys_update_btn").addClass('hide');
	$("#hide_sys_update_btn").removeClass('hide');
	
	//CHECKING ALL THE REMARK TYPES
	var rem_type_count = 0;
	$.each($("#remark_types_dropdown li"), function(){
		var rem_type = $.trim($(this).text());
		if(!$('#remark_types_dropdown [value="'+rem_type+'"]').is(':checked')) {
			$('#remark_types_dropdown [value="'+rem_type+'"]').prop('checked',true);
		}
		if(rem_type != '' && rem_type != 'MORE'){
			rem_type_count++;
		}
	});
	
 	$("#remark_filter_div").addClass('hide');
//	$("#list_heading").html('All Remark Types');
    $("#filter_remark_type").val('');
    $("#filter_remark_type").attr('remark_type_count',rem_type_count);
	$('#remark_filter_div').addClass('hide');

	if(from_page == 'ACCOUNT_DETAIL') {
		LoadMoreDetailSysUdates(15, 'history_timeline', 'more_history_btn');
	}
	else{
		LoadMoreAccountSysUdates(15, 'history_timeline', 'more_history_btn');
	}
}

function HideSystemUpdates() {
	HidePopover();
	$("#sys_updates").val('0');
	$("#filter_remark_type").val('');
	$('#history_timeline').html('');
	$('#history_timeline').attr('limit_from',0);
	$('#more_history_btn').show();

	$("#show_sys_update_btn").removeClass('hide');
	$("#hide_sys_update_btn").addClass('hide');
	
	var rem_type_count = 0;
	
	// UN CHECKING ALL THE INTERNAL REMARK TYPES 
	$.each($("#remark_types_dropdown li"), function(){
		var rem_type = $.trim($(this).text());
		if(rem_type != 'MORE' && rem_type != ''){ 
        	var rem_index = remark_filter.indexOf(rem_type);
        	
    		if($(this).hasClass('sys_remarks')){
        		if($('#remark_types_dropdown [value="'+rem_type+'"]').is(':checked')) {
        			$('#remark_types_dropdown [value="'+rem_type+'"]').prop('checked',false);
        		}
				
				//IF SYSTEM REMARK TYPE ALREADY AVAILABLE IN THE ARRAY, REMOVING IN THE REMARK FILTER ARRAY
                if (rem_index !== -1) {
                    remark_filter.splice(rem_index, 1);
            	}
        		$(this).addClass('hide');
        	}
        	else {
            	rem_type_count++;
        		if(rem_index == -1){
                	remark_filter.push(rem_type);
                	$('#remark_types_dropdown [value="'+rem_type+'"]').prop('checked',true);
                }
            }
    	}
    	
	});
    $("#filter_remark_type").val(JSON.stringify(remark_filter));
 	$("#remark_filter_div").addClass('hide');
//	$("#list_heading").html('Remark Types within '+ remark_filter.join(", "));
    $("#filter_remark_type").attr('remark_type_count',rem_type_count);
	$('#more_options').removeClass('hide');

	if(from_page == 'ACCOUNT_DETAIL') {
		LoadMoreDetailSysUdates(15, 'history_timeline', 'more_history_btn');
	}
	else{
		LoadMoreAccountSysUdates(15, 'history_timeline', 'more_history_btn');
	}
}

function CheckExportRemarksToExcel(){
	var account_no = $("#account_number").val(),
		include_sys_updates = parseInt($('#sys_updates').val());
		var selected_remark_types = $('#filter_remark_type').val();
		var selected_remark_types_count = parseInt($('#filter_remark_type').attr('remark_type_count'));
		
		if(selected_remark_types_count > 0) {
            $.ajax({
                    type: "POST",
                    url: "/core/ajax_manage_account.php",
                    data: {account_no:account_no,include_sys_updates:include_sys_updates,obj:'remark',action:'check_remarks_row_limit', filter: selected_remark_types},
                    dataType: 'json',
                    success: function(data) { 
                        if(data.RESPONSE == 'SUCCESS'){
                        	ExportRemarksToExcel('ACCOUNT', account_no, include_sys_updates, data.TOTAL_ROW_COUNT, selected_remark_types); //Expected format ExportRemarksToExcel(context='LEAD', context_no='', include_sys_updates=0, total_row_count=0, selected_remark_types)
                        }
                        else if(data.RESPONSE == 'LIMIT_EXCEEDED'){
                            var max_row_limit = $('#max_report_rows').val();
                            bootbox.confirm({
                                message: "Your export request has resulted in more than the allowed limit of rows("+max_row_limit+"). So it will export Only the latest "+max_row_limit+" rows in excel.",
                                callback: function(result) {
                                            if(result) {
                                            	ExportRemarksToExcel('ACCOUNT', account_no, include_sys_updates, data.TOTAL_ROW_COUNT, selected_remark_types); //Expected format ExportRemarksToExcel(context='LEAD', context_no='', include_sys_updates=0, total_row_count=0, selected_remark_types)
                                            }
                                        }
                            });
                        }
                        else if(data.RESPONSE == 'ERROR'){
                            // ERROR GREET MESSAGE
                        }
                    }
            });
      }
      else {
    	  bootbox.alert('Please select a remark type to export to excel.');
      }
}
